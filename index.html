Panduan Setup Google Sheets Sync

Ikuti langkah berikut untuk mengaktifkan sinkronisasi dua arah dengan Google Sheets.

1Buat Google Sheet Baru

Buka sheets.new → buat spreadsheet baru dengan nama Offset Analyzer DB

2Buka Apps Script

Di Google Sheet → klik Extensions → Apps Script

3Paste Script di bawah ini

Hapus semua isi editor, paste script berikut, lalu klik Save:

// ═══════════════════════════════════════════════ // OFFSET ANALYZER PRO - Google Apps Script // Salin seluruh kode ini ke Apps Script Editor // ═══════════════════════════════════════════════ const SHEET_NAME = "OffsetDB"; const SECRET_KEY = "OFFSET_PRO_2024"; // Ganti jika perlu function doGet(e) { try { const sheet = getOrCreateSheet(); const data = readAllData(sheet); return ContentService.createTextOutput(JSON.stringify({ status: "ok", db: data })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const payload = JSON.parse(e.postData.contents); if (payload.key !== SECRET_KEY) { return ContentService.createTextOutput(JSON.stringify({ status: "unauthorized" })) .setMimeType(ContentService.MimeType.JSON); } const sheet = getOrCreateSheet(); if (payload.action === "SAVE_ALL") { writeAllData(sheet, payload.db); return ContentService.createTextOutput(JSON.stringify({ status: "saved" })) .setMimeType(ContentService.MimeType.JSON); } if (payload.action === "UPDATE_ROW") { updateRow(sheet, payload.item); return ContentService.createTextOutput(JSON.stringify({ status: "updated" })) .setMimeType(ContentService.MimeType.JSON); } } catch(err) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function getOrCreateSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["line","id","type","op","tool","x","z","updated_at"]); sheet.getRange(1,1,1,8).setBackground("#1a1a2e").setFontColor("#fbbf24").setFontWeight("bold"); sheet.setFrozenRows(1); } return sheet; } function readAllData(sheet) { const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return { line1:[], line2:[], line3:[], line4:[] }; const db = { line1:[], line2:[], line3:[], line4:[] }; for (let i = 1; i < rows.length; i++) { const [line, id, type, op, tool, x, z] = rows[i]; if (!line || !id) continue; const key = "line" + line; if (!db[key]) db[key] = []; db[key].push({ id: String(id), type: String(type), op: String(op), tool: String(tool), x: parseFloat(x)||0, z: parseFloat(z)||0 }); } return db; } function writeAllData(sheet, db) { const lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.deleteRows(2, lastRow - 1); const rows = []; for (let ln = 1; ln <= 4; ln++) { const items = db["line" + ln] || []; items.forEach(item => { rows.push([ln, item.id, item.type, item.op, item.tool, item.x, item.z, new Date().toISOString()]); }); } if (rows.length > 0) sheet.getRange(2, 1, rows.length, 8).setValues(rows); } function updateRow(sheet, item) { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][1]) === String(item.id)) { sheet.getRange(i+1, 1, 1, 8).setValues([[item.line, item.id, item.type, item.op, item.tool, item.x, item.z, new Date().toISOString()]]); return; } } sheet.appendRow([item.line, item.id, item.type, item.op, item.tool, item.x, item.z, new Date().toISOString()]); }
4Deploy sebagai Web App

Klik Deploy → New deployment → pilih type Web app → Execute as: Me → Who has access: Anyone → klik Deploy

5Salin Web App URL

Salin URL yang muncul (format: https://script.google.com/macros/s/.../exec) → paste di field URL pada Admin Panel → klik Koneksikan
